async function fetchSummary() { console.log("🔥 fetchSummary() function triggered!"); const topic = document.getElementById("topicInput").value.trim(); const resultDiv = document.getElementById("result"); if (!topic) { resultDiv.innerHTML = "

Please enter a topic!

"; return; } resultDiv.innerHTML = "

Loading...

"; const BASE_URL = window.location.hostname.includes("localhost") ? "http://127.0.0.1:8000" : "https://azukimedia.up.railway.app"; try { console.log(`🔗 Fetching: ${BASE_URL}/summary/${encodeURIComponent(topic)}`); const response = await fetch(`${BASE_URL}/summary/${encodeURIComponent(topic)}`, { method: "GET", headers: { "Accept": "application/json" } }); console.log("🟢 Response Status:", response.status); if (!response.ok) { throw new Error(`❌ Server error: ${response.status} - ${response.statusText}`); } const data = await response.json(); console.log("✅ API Response:", data); // ✅ Log the full response // 🔹 Ensure `data.summary` is correctly displayed let summaryText = "No summary available."; if (typeof data.summary === "string") { summaryText = data.summary.replace(/\n/g, "
"); // ✅ Convert line breaks to HTML } else if (data.summary && typeof data.summary.raw === "string") { summaryText = data.summary.raw.replace(/\n/g, "
"); // ✅ Handle nested object case } resultDiv.innerHTML = `

📢 Latest "${topic}" Updates:

${summaryText}
`; } catch (error) { console.error("🚨 Fetch error:", error); resultDiv.innerHTML = `

Error fetching summary: ${error.message}. Try again!

`; } }
